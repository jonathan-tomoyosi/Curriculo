"use client";
import Link from "next/link";
import { useState } from "react";
import LoginModal from "../components/LoginModal";
import { useAuth } from "@/context/AuthContext"; // importa o contexto
import {
  FaBriefcase,
  FaCalendarAlt,
  FaQrcode,
  FaMapMarkerAlt,
  FaInstagram,
} from "react-icons/fa";
import { FaUser } from "react-icons/fa6";
import { TbLogout2 } from "react-icons/tb";
import { PiHandsPrayingFill } from "react-icons/pi";


export default function HomePage() {
  const [showModal, setShowModal] = useState(false);
  const { user, logout } = useAuth();

  // Pega o primeiro nome do user_metadata.nome ou do email
  const primeiroNome =
    user?.user_metadata?.nome?.split(" ")[0] ||
    user?.email?.split("@")[0] ||
    "";

  return (
    <main>
      {/* Se não tiver usuário logado, mostra botão de login */}
      {!user ? (
        <button className="login-button" onClick={() => setShowModal(true)}>
          <FaUser />
        </button>
      ) : (
        <div className="auth-info">
          <p>Bem-vindo{'(a)'}, {primeiroNome}!</p>
          <button className="logout-button" onClick={logout}>
            <TbLogout2 />
          </button>
        </div>
      )}

      <div className="logo">
        <picture>
          <img
            src="/bom_pastor_processed.png"
            alt="Logo Cordeiro"
            className="logo-img"
          />
        </picture>
      </div>

      <nav className="menu">
        <Link href="/Operarios" className="botao">
          <FaBriefcase className="icon" />
          <p>Operários</p>
        </Link>

        <Link href="/Calendario" className="botao">
          <FaCalendarAlt className="icon" />
          <span>Eventos</span>
        </Link>

        <Link href="/Oracao" className="botao">
          <PiHandsPrayingFill className="icon"/>
          <span>Pedido de oração</span>
        </Link>

        <Link href="/Dizimo" className="botao">
          <FaQrcode className="icon" />
          <span>Dízimo e Ofertas</span>
        </Link>

        <Link
          href="https://www.instagram.com/bompastorsp"
          target="_blank"
          className="botao"
        >
          <FaInstagram className="icon" />
          <span>Instagram</span>
        </Link>

        <Link 
          href="https://waze.com/ul?q=R.%20Lisboa,%2085%20-%20Oswaldo%20Cruz%2C%20São%20Caetano%20do%20Sul%20-%20SP%2C%2009570-510"
          target="_blank"
          className="botao">
          <FaMapMarkerAlt className="icon" />
          <span>Endereço</span>
        </Link>
      </nav>

      {showModal && <LoginModal onClose={() => setShowModal(false)} />}
    </main>
  );
}
